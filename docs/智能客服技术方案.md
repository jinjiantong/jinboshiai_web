### 3.3 同步代码实现

#### 3.3.1 MySQL → Qdrant 同步服务

```typescript
// lib/sync-to-vector.ts
import { vectorDB } from './vector-db'
import { getEmbedding } from './embedding'
import { getDb } from './db'

const COLLECTION_NAME = 'chatbot_knowledge'

interface KnowledgeItem {
  id: number
  title: string
  content: string
  category: string
  sub_category: string
  keywords: string[]
  status: number
  updated_at: Date
}

export async function syncToVectorDb(options: {
  fullSync?: boolean
  since?: Date
  batchSize?: number
} = {}): Promise<{ synced: number; failed: number }> {
  const { fullSync = false, since, batchSize = 50 } = options
  
  console.log(`[Sync] Starting sync: fullSync=${fullSync}, since=${since}`)
  
  const db = await getDb()
  
  // 1. 查询需要同步的数据
  let query = 'SELECT * FROM knowledge_base WHERE status = 1'
  if (!fullSync && since) {
    query += ` AND updated_at > '${since.toISOString()}'`
  }
  query += ' ORDER BY updated_at DESC'
  
  const [rows] = await db.query<any[]>(query)
  console.log(`[Sync] Found ${rows.length} items to sync`)
  
  let synced = 0
  let failed = 0
  
  // 2. 批量处理
  for (let i = 0; i < rows.length; i += batchSize) {
    const batch = rows.slice(i, i + batchSize)
    
    const vectors = []
    for (const row of batch) {
      try {
        const text = `${row.title}\n\n${row.content}`
        const vector = await getEmbedding(text)
        
        if (vector.length > 0) {
          vectors.push({
            id: row.id,
            text,
            vector,
            metadata: {
              title: row.title,
              category: row.category,
              subCategory: row.sub_category,
              keywords: row.keywords ? JSON.parse(row.keywords) : []
            }
          })
        }
      } catch (error) {
        console.error(`[Sync] Failed to generate vector for item ${row.id}:`, error)
        failed++
      }
      
      await new Promise(r => setTimeout(r, 100))
    }
    
    // 3. 批量写入 Qdrant
    if (vectors.length > 0) {
      try {
        await vectorDB.upsert(COLLECTION_NAME, vectors)
        synced += vectors.length
        console.log(`[Sync] Synced batch: ${vectors.length} items`)
      } catch (error) {
        console.error(`[Sync] Failed to insert batch:`, error)
        failed += vectors.length
      }
    }
  }
  
  console.log(`[Sync] Complete: synced=${synced}, failed=${failed}`)
  return { synced, failed }
}
```

#### 3.3.2 同步状态记录

```typescript
// lib/sync-status.ts
import { getDb } from './db'

export async function recordSyncStart(
  syncType: string,
  syncMode: string
): Promise<number> {
  const db = await getDb()
  const [result] = await db.query(
    `INSERT INTO sync_history (sync_type, sync_mode, status, started_at)
     VALUES (?, ?, 'running', NOW())`,
    [syncType, syncMode]
  )
  return (result as any).insertId
}

export async function recordSyncComplete(
  id: number,
  total: number,
  synced: number,
  failed: number,
  error?: string
): Promise<void> {
  const db = await getDb()
  await db.query(
    `UPDATE sync_history SET 
       status = ?,
       completed_at = NOW(),
       total_items = ?,
       synced_items = ?,
       failed_items = ?,
       error_message = ?
     WHERE id = ?`,
    [
      error ? 'failed' : 'completed',
      total, synced, failed, error, id
    ]
  )
}

export async function getLastSyncTime(syncType: string): Promise<Date | null> {
  const db = await getDb()
  const [rows] = await db.query(
    `SELECT completed_at FROM sync_history 
     WHERE sync_type = ? AND status = 'completed'
     ORDER BY completed_at DESC LIMIT 1`,
    [syncType]
  )
  return rows.length > 0 ? rows[0].completed_at : null
}
```

### 3.6 同步 API 接口

```typescript
// app/api/sync/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { syncToVectorDb } from '@/lib/sync-to-vector'
import { recordSyncStart, recordSyncComplete, getLastSyncTime } from '@/lib/sync-status'

export async function POST(request: NextRequest) {
  try {
    const { type, mode } = await request.json()
    
    if (type === 'mysql_to_qdrant') {
      const since = mode === 'incremental' 
        ? await getLastSyncTime('mysql_to_qdrant')
        : null
      
      const syncId = await recordSyncStart('mysql_to_qdrant', mode)
      
      const result = await syncToVectorDb({
        fullSync: mode === 'full',
        since: since || undefined
      })
      
      await recordSyncComplete(
        syncId,
        result.synced + result.failed,
        result.synced,
        result.failed
      )
      
      return NextResponse.json({
        success: true,
        type: 'mysql_to_qdrant',
        mode,
        result
      })
    }
    
    return NextResponse.json({
      success: false,
      message: 'Invalid sync type'
    }, { status: 400 })
    
  } catch (error) {
    console.error('Sync error:', error)
    return NextResponse.json({
      success: false,
      message: 'Sync failed'
    }, { status: 500 })
  }
}

export async function GET() {
  const db = await getDb()
  const [rows] = await db.query(
    `SELECT * FROM sync_history ORDER BY started_at DESC LIMIT 10`
  )
  
  return NextResponse.json({
    success: true,
    history: rows
  })
}
```

### 5.1 Qdrant 客户端

```typescript
// lib/vector-db.ts
interface VectorItem {
  id: number
  text: string
  vector: number[]
  metadata?: Record<string, any>
}

class QdrantClient {
  private baseUrl: string

  constructor(host: string, port: number = 6333) {
    this.baseUrl = `http://${host}:${port}`
  }

  async createCollection(name: string, dimension: number = 1024): Promise<void> {
    await fetch(`${this.baseUrl}/collections/${name}`, {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ vectors: { size: dimension, distance: 'Cosine' } })
    })
  }

  async upsert(name: string, vectors: VectorItem[]): Promise<void> {
    await fetch(`${this.baseUrl}/collections/${name}/points?wait=true`, {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        points: vectors.map((v, i) => ({
          id: v.id || i + 1,
          vector: v.vector,
          payload: { text: v.text, ...v.metadata }
        }))
      })
    })
  }

  async search(name: string, vector: number[], topK: number = 3): Promise<any[]> {
    const response = await fetch(`${this.baseUrl}/collections/${name}/points/search`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ vector, limit: topK, with_payload: true, score_threshold: 0.5 })
    })
    const data = await response.json()
    return data.result || []
  }

  async exists(name: string): Promise<boolean> {
    const response = await fetch(`${this.baseUrl}/collections/${name}`)
    return response.ok
  }
}

export const vectorDB = new QdrantClient(
  process.env.QDRANT_HOST || 'localhost',
  parseInt(process.env.QDRANT_PORT || '6333')
)
```

```typescript
import crypto from 'crypto'

const config = {
  apiKey: process.env.MINIMAX_API_KEY || '',
  groupId: process.env.MINIMAX_GROUP_ID || '',
  baseUrl: 'https://api.minimax.chat/v1'
}

function generateSignature(): string {
  const timestamp = Math.floor(Date.now() / 1000)
  return crypto.createHmac('sha256', config.apiKey)
    .update(`${config.groupId}{${timestamp}}`).digest('hex')
}

export async function getEmbedding(text: string): Promise<number[]> {
  const timestamp = Math.floor(Date.now() / 1000)
  const signature = generateSignature()
  const response = await fetch(`${config.baseUrl}/embeddings`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${config.apiKey}` },
    body: JSON.stringify({ model: 'embo-01', texts: [text] })
  })
  const data = await response.json()
  return data.data?.[0]?.embedding || []
}

export async function chatCompletion(
  messages: Array<{ role: string; content: string }>,
  options?: { temperature?: number; maxTokens?: number }
): Promise<{ content: string; tokens: number }> {
  const timestamp = Math.floor(Date.now() / 1000)
  const signature = generateSignature()
  const response = await fetch(`${config.baseUrl}/text/chatcompletion_v2`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${config.apiKey}` },
    body: JSON.stringify({
      model: 'abab6.5s',
      tokens_to_generate: options?.maxTokens || 1024,
      temperature: options?.temperature || 0.7,
      messages: messages.map(m => ({ sender_type: m.role === 'assistant' ? 'bot' : m.role, text: m.content })),
      group_id: config.groupId, signature, timestamp
    })
  })
  const data = await response.json()
  const assistantMsg = data.choices?.[0]?.messages?.find((m: any) => m.sender_type === 'bot')
  return { content: assistantMsg?.text || '', tokens: data.usage?.tokens || 0 }
}
```

```typescript
const SYSTEM_PROMPT = `你是一个专业的金博士AI课程咨询助手。

    const { message, sessionId, history } = await request.json()
    if (!message) return NextResponse.json({ success: false, message: '请输入问题' }, { status: 400 })

    const ragResult = await retrieve(message)
    const contextPrompt = ragResult.found && ragResult.context
      ? `\n\n【参考资料】\n${ragResult.context}\n\n请基于以上参考资料回答用户问题。`
      : ''

    const response = await chatCompletion([
      { role: 'system', content: SYSTEM_PROMPT + contextPrompt },
      ...(history || []).slice(-6),
      { role: 'user', content: message }
    ])

    return NextResponse.json({
      success: true,
      message: response.content,
      sources: ragResult.sources,
      tokens: response.tokens
    })
  } catch (error) {
    return NextResponse.json({ success: false, message: '服务暂时不可用' }, { status: 500 })
  }
}
```
