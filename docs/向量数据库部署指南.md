> 使用腾讯云轻量应用服务器（Ubuntu）部署 Qdrant 向量数据库

## 二、Qdrant 部署（推荐）

### 2.1 环境要求

- Ubuntu 20.04+ / 22.04+
- 1核 1GB+ RAM
- 10GB+ SSD

### 2.2 Docker 部署（推荐）

```bash
# 1. 安装 Docker
sudo apt update
sudo apt install -y docker.io docker-compose

# 2. 启动 Docker 服务
sudo systemctl start docker
sudo systemctl enable docker

# 3. 创建 Qdrant 目录
mkdir -p ~/qdrant && cd ~/qdrant

# 4. 创建 docker-compose.yml
cat > docker-compose.yml << 'EOF'
version: '3.8'

services:
  qdrant:
    image: qdrant/qdrant:latest
    container_name: qdrant
    ports:
      - "6333:6333"      # REST API
      - "6334:6334"      # gRPC API
    volumes:
      - ./qdrant_storage:/qdrant/storage
    environment:
      - QDRANT__SERVICE__GRPC_PORT=6334
      - QDRANT__SERVICE__MAX_REQUEST_SIZE_MB=32
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:6333/readyz"]
      interval: 10s
      timeout: 5s
      retries: 5
EOF

# 5. 启动服务
sudo docker-compose up -d

# 6. 验证服务
curl http://localhost:6333/readyz
```

### 2.3 非 Docker 部署（备选）

```bash
# 下载 Qdrant
wget https://github.com/qdrant/qdrant/releases/latest/download/qdrant-linux-amd64.tar.gz
tar -xzf qdrant-linux-amd64.tar.gz

# 启动服务（前台运行）
./qdrant

# 或后台运行
nohup ./qdrant --storage-path ./storage > qdrant.log 2>&1 &
```

### 2.4 服务验证

```bash

# 查看健康状态（返回 ok 表示成功）
```

## 四、代码集成

### 4.2 TypeScript 客户端封装

```typescript
// lib/vector-db-qdrant.ts

interface QdrantConfig {
  host: string
  port: number
}

interface VectorItem {
  id: string | number
  text: string
  vector: number[]
  metadata?: Record<string, any>
}

interface SearchResult {
  id: string | number
  score: number
  text: string
  metadata: Record<string, any>
}

class QdrantVectorDB {
  private baseUrl: string

  constructor(config: QdrantConfig = { host: 'YOUR_SERVER_IP', port: 6333 }) {
    this.baseUrl = `http://${config.host}:${config.port}`
  }

  async createCollection(
    name: string,
    dimension: number = 1024
  ): Promise<void> {
    const response = await fetch(`${this.baseUrl}/collections/${name}`, {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        vectors: {
          size: dimension,
          distance: 'Cosine'
        }
      })
    })

    if (!response.ok) {
      throw new Error(`Failed to create collection: ${response.status}`)
    }
  }

  async insert(
    collectionName: string,
    vectors: VectorItem[]
  ): Promise<void> {
    const points = vectors.map(v => ({
      id: typeof v.id === 'string' ? parseInt(v.id) : v.id,
      vector: v.vector,
      payload: {
        text: v.text,
        ...v.metadata
      }
    }))

    const response = await fetch(
      `${this.baseUrl}/collections/${collectionName}/points?wait=true`,
      {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ points })
      }
    )

    if (!response.ok) {
      throw new Error(`Failed to insert vectors: ${response.status}`)
    }
  }

  async search(
    collectionName: string,
    queryVector: number[],
    options: { topK?: number } = {}
  ): Promise<SearchResult[]> {
    const { topK = 3 } = options

    const response = await fetch(
      `${this.baseUrl}/collections/${collectionName}/points/search`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          vector: queryVector,
          limit: topK,
          with_payload: true
        })
      }
    )

    if (!response.ok) {
      throw new Error(`Failed to search vectors: ${response.status}`)
    }

    const data = await response.json()
    
    return (data.result || []).map((item: any) => ({
      id: item.id,
      score: item.score,
      text: item.payload?.text || '',
      metadata: item.payload || {}
    }))
  }

  async deleteCollection(name: string): Promise<void> {
    const response = await fetch(`${this.baseUrl}/collections/${name}`, {
      method: 'DELETE'
    })

    if (!response.ok) {
      throw new Error(`Failed to delete collection: ${response.status}`)
    }
  }

  async getCollectionInfo(name: string): Promise<any> {
    const response = await fetch(`${this.baseUrl}/collections/${name}`)
    const data = await response.json()
    return data.result
  }
}

export const vectorDB = new QdrantVectorDB({
  host: process.env.QDRANT_HOST || 'localhost',
  port: 6333
})

export { VectorItem, SearchResult }
```

### 4.3 完整 RAG 实现

```typescript
// lib/rag.ts
import { vectorDB } from './vector-db-qdrant'
import { getEmbedding } from './embedding'

const COLLECTION_NAME = 'chatbot_knowledge'
const TOP_K = 3

interface RAGResult {
  context: string
  sources: Array<{
    title: string
    category: string
    score: number
  }>
}

export async function retrieve(
  query: string,
  options: { topK?: number } = {}
): Promise<RAGResult> {
  const { topK = TOP_K } = options

  const queryVector = await getEmbedding(query)
  
  if (queryVector.length === 0) {
    return { context: '', sources: [] }
  }

  try {
    const results = await vectorDB.search(COLLECTION_NAME, queryVector, { topK })

    if (results.length === 0) {
      return { context: '', sources: [] }
    }

    const context = results
      .map((r, i) => `[参考${i + 1}] ${r.text}`)
      .join('\n\n')

    const sources = results.map(r => ({
      title: r.metadata?.title || '未知',
      category: r.metadata?.category || '其他',
      score: r.score
    }))

    return { context, sources }
  } catch (error) {
    console.error('RAG retrieve error:', error)
    return { context: '', sources: [] }
  }
}

export async function initKnowledgeBase(knowledgeItems: any[]): Promise<void> {
  console.log('Initializing knowledge base...')

  try {
    await vectorDB.createCollection(COLLECTION_NAME, 1024)
  } catch (e) {
    console.log('Collection may already exist')
  }

  const vectors = await Promise.all(
    knowledgeItems.map(async (item, index) => {
      const text = `${item.title}\n\n${item.text}`
      const vector = await getEmbedding(text)
      return {
        id: index + 1,
        text,
        vector,
        metadata: {
          category: item.category,
          subCategory: item.subCategory,
          title: item.title,
          keywords: item.keywords
        }
      }
    })
  )

  await vectorDB.insert(COLLECTION_NAME, vectors.filter(v => v.vector.length > 0))
  console.log(`Knowledge base initialized with ${vectors.length} items`)
}
```

## 六、Nginx 反向代理 + 认证

```nginx
# /etc/nginx/conf.d/qdrant.conf

server {
    listen 6333;
    server_name YOUR_SERVER_IP;

    location / {
        # Basic Auth
        auth_basic "Qdrant Vector DB";
        auth_basic_user_file /etc/nginx/.htpasswd;

        # 代理到 Qdrant
        proxy_pass http://localhost:6333;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
}

# 生成密码文件
sudo apt install -y apache2-utils
sudo htpasswd -bc /etc/nginx/.htpasswd admin YOUR_PASSWORD
```

## 七、性能优化

### 7.1 HNSW 参数调优

```json
{
  "optimize": {
    "indexing": {
      "vector_index": {
        "type": "HNSW",
        "params": {
          "m": 16,           // 节点连接数，越大越精确但越慢
          "ef_construct": 200  // 构建时搜索深度，越大越精确
        }
      }
    }
  }
}
```

### 7.2 内存配置

Qdrant 默认配置适合中小规模数据。如需优化，修改 `config.yaml`：

```yaml
# ~/.qdrant/config.yaml
storage:
  on_disk_payload: false  # 关闭磁盘存储，全部加载到内存
```
